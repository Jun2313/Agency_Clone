네, 이 문제를 해결하려면 position: fixed 속성과 함께 top: 100px를 사용하는 것이 맞습니다. 스크롤이 일어날 때 nav 부분이 위로 움직이는 것을 방지하기 위해, LayoutNav.js 컴포넌트의 최상위 div에 고정 높이와 오버플로우 속성을 추가해야 합니다. 이렇게 하면 nav 부분 내부에서만 스크롤이 일어나고, 외부 요소에는 영향을 주지 않게 됩니다. 여기에 코드를 보여드리겠습니다:

javascript
function Nav({isOpen}) {
  return (
    <>
      <div style={{ height: 'calc(100% - 100px)', overflowY: 'auto', top: '100px', position: 'fixed' }}>
        <div id="hamburger-menu" className={isOpen ? 'open' : ''}>
          {/* 나머지 코드는 그대로 유지 */}
        </div>
      </div>
    </>
  );
}

위 코드에서 height: 'calc(100% - 100px)'는 화면의 전체 높이에서 100px (header의 높이)를 뺀 나머지 높이를 nav 영역의 높이로 설정합니다. overflowY: 'auto'는 이 영역 내부에서만 스크롤이 가능하게 합니다. 그리고 top: '100px', position: 'fixed' 속성을 통해 nav 영역이 항상 header 아래에 위치하게 합니다.

CSS 부분에서도 #hamburger-menu의 높이를 조절해야 합니다:

css
#hamburger-menu {
  height: calc(100% - 100px); /* 여기를 변경 */
  /* 나머지 속성은 그대로 유지 */
}

이러한 변경을 통해 원하시는 기능을 얻으실 수 있을 것입니다.

css
#hamburger-menu {

function ContextAwareToggle({ children, eventKey, callback }) {
  const { activeEventKey } = useContext(AccordionContext);

  const decoratedOnClick = useAccordionButton(
    eventKey,
    () => callback && callback(eventKey),
  );

  const isCurrentEventKey = activeEventKey === eventKey;

  return (
    <button
      type="button"
      style={{ backgroundColor: isCurrentEventKey ? 'transparent' : 'transparent' }}
      onClick={decoratedOnClick}
    >
      {children}
    </button>
  );
}

function ContextAwareToggle({ children, eventKey, callback }) {
  const { activeEventKey } = useContext(AccordionContext);
